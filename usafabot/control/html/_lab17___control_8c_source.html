84 // ------------avg------------
85 // Simple math function that returns the average
86 // value of an array.
87 // Input: array is an array of 16-bit unsigned numbers
88 // length is the number of elements in 'array'
89 // Output: the average value of the array
90 // Note: overflow is not considered
91 uint16_t avg(uint16_t *array, int length){
92  int i;
93  uint32_t sum = 0;
94  for(i=0; i<length; i=i+1){
95  sum = sum + array[i];
96  }
97  return (sum/length);
98 }

103 // Integral control, Line follower
104 void Controller1(void){
105  // write this as part of Lab 17
106  if(Mode){
107  Tachometer_Get(&LeftTach[i], &LeftDir, &LeftSteps, &RightTach[i], &RightDir, &RightSteps);
108  i = i + 1;
109  if(i >= TACHBUFF){
110  i = 0;
111  XstarL = DESIRED_SPEED;
112  XstarR = DESIRED_SPEED;
113  XprimeL = 2000000/avg(LeftTach, TACHBUFF);
114  XprimeR = 2000000/avg(RightTach, TACHBUFF);
115  ErrorL = XstarL - XprimeL;
116  ErrorR = XstarR - XprimeR;
117  UL = UL + (A*ErrorL)/1024; // calculate integral
118  UR = UR + (A*ErrorR)/1024; // calculate integral
119  if (UL < 2) UL = 2;
120  if (UR < 2) UR = 2;
121  if (UL > 14998) UL = 14998;
122  if (UR > 14998) UR = 14998;
123  Motor_Forward(UL, UR);
124  }
125  }
126 }

261 // record 32 halfwords
262 void Debug_FlashRecord(int32_t *pt){
263  uint32_t addr;
264  addr=FLASH_BANK1_MIN;
265  while(*(uint32_t*)addr != 0xFFFFFFFF){ // find first free block
266  addr=addr+64;
267  if(addr>FLASH_BANK1_MAX) return; // full
268  }
269  Flash_FastWrite((uint32_t *)pt, addr, 16); // 16 words is 32 halfwords, 64 bytes
270 }

272 void SysTick_Handler(void){
273  if(Mode){
274  //
275  if((Left>DESIRED_DIST)&&(Right>DESIRED_DIST))
276  {
277  SetPoint = (Left+Right)/2;
278  }else{
279  SetPoint = DESIRED_DIST;
280  }
281  //
282  if(Left < Right){
283  Error = Left-SetPoint;
284  }else{
285  Error = SetPoint-Right;
286  }
287  UR = PWMNOMINAL+Kp*Error;
288  UL = PWMNOMINAL-Kp*Error;
289  if(UR < (PWMNOMINAL-SWING)) UR = PWMIN;
290  if(UR > (PWMNOMINAL+SWING)) UR = PWMAX;
291  if(UL < (PWMNOMINAL-SWING)) UL = PWMIN;
292  if(UL > (PWMNOMINAL+SWING)) UL = PWMAX;
293  Motor_Forward(UL, UR);
294  ControllerFlag = 1;
295  DataBuffer[0] = Left;
296  DataBuffer[1] = Right;
297  DataBuffer[2] = Error;
298  DataBuffer[3] = UL;
299  DataBuffer[4] = UR;
300  Debug_FlashRecord(&DataBuffer[0]);
301  }
302 }
